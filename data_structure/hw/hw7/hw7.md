```c++
void delete_e(int e)
{
    int num = hash(e);
    while (hash_table[num] != e)
    {
        num = (num + 1) % 10;
        if (hash_table[num] == 0)
            break;
    }

}
```

```c++
struct BiTree
{
    int val;
    struct BiTree *left, *right;
};

bool judge(struct BiTree *H)
{
    bool ans = true;
    if (H->left != nullptr)
        ans = ans && H->left->val < H->val && judge(H->left);
    if (H->right != nullptr)
        ans = ans && H->right->val > H->val && judge(H->right);
    return ans;
}
```
